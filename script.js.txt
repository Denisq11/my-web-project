function addTask() {
    const input = document.getElementById('taskInput');
    const taskText = input.value.trim();
    
    if (taskText !== "") {
        const list = document.getElementById('taskList');
        const li = document.createElement('li');
        
        li.innerHTML = `
            <span>${taskText}</span>
            <button onclick="this.parentElement.remove()" style="background:transparent; color:red; padding:0">✕</button>
        `;
        
        list.appendChild(li);
        input.value = ""; // Очистити поле
    } else {
        alert("Будь ласка, введіть текст завдання!");
    }
}
